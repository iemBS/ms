Note:
  -Guidance from https://docs.microsoft.com/en-us/sql/t-sql/queries/table-value-constructor-transact-sql?view=sql-server-2017

Declare @xyz table(
abc varchar(10),
pqr varchar(10)
)

insert into @xyz(abc, pqr)
select abc, pqr
from (VALUES
    ('a1', 'p1'),
    ('a2', 'p2'),
    ('a3', 'p3')
)t(abc, pqr)

select * from @xyz
