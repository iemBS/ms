Get Distinct Column Value Counts in Table
Note:
  -Note if column is used an index or key, data type, nullable, precision, scale, length, & default value
Main Success Scenario:
  1.see below

declare @col varchar(40)
declare @qry varchar(200)
declare @sch varchar(40)
declare @tbl varchar(40)

set @sch = 'mySchema'
set @tbl = 'myTable'

drop table if exists #col

select top 10
	*
from
	INFORMATION_SCHEMA.COLUMNS

select
	column_name,
	IS_NULLABLE,
	DATA_TYPE,
	NUMERIC_PRECISION,
	NUMERIC_SCALE,
	CHARACTER_MAXIMUM_LENGTH
into
	#col
from
	INFORMATION_SCHEMA.COLUMNS
where
	table_schema = @sch
	and
	table_name = @tbl

set @qry = 'select count(1) as tblRowCnt from ' + @sch + '.' + @tbl + ' with(nolock)'

execute(@qry)

while exists(select top 1 * from #col)
begin
	select top 1
		@col = column_name
	from
		#col

	set @qry = 'select ''' + @col + ''' as colName,count(distinct ' + @col + ') as distinctValCnt from ' + @sch + '.' + @tbl + ' with(nolock)'

	execute(@qry)

	delete #col where column_name = @col
end
